Buried particles

Scattering from a sample containing spherical embedded particles.

  • From top to bottom, the sample is composed of the air layer, an intermediate material layer, and the substrate.
  • The particles are 10.2-nanometer-radius spheres.
  • They are placed vertically in the middle of the intermediate layer. The depth to place the spheres is measured between the top of the layer and the bottom of the particles.
  • There is no interference between the scattered waves. The horizontal distribution of the particles is diluted.
  • The wavelength is equal to 1.5 Å.
  • The incident angles are αi = 0.15° and Φi = 0°.
 1"""
 2Spherical particles embedded in the middle of the layer on top of substrate.
 3"""
 4import bornagain as ba
 5from bornagain import deg, angstrom, nm
 6
 7
 8def get_sample():
 9    """
10    Returns a sample with spherical particles in an layer between air and substrate.
11    """
12    # defining materials
13    m_ambience = ba.HomogeneousMaterial("Air", 0.0, 0.0)
14    m_interm_layer = ba.HomogeneousMaterial("IntermLayer", 3.45e-6, 5.24e-9)
15    m_substrate = ba.HomogeneousMaterial("Substrate", 7.43e-6, 1.72e-7)
16    m_particle = ba.HomogeneousMaterial("Particle", 0.0, 0.0)
17
18    # collection of particles
19    ff_sphere = ba.FormFactorFullSphere(10.2*nm)
20    sphere = ba.Particle(m_particle, ff_sphere)
21    sphere.setPosition(0.0, 0.0, -25.2)
22    particle_layout = ba.ParticleLayout()
23    particle_layout.addParticle(sphere, 1.0)
24
25    # assembling the sample
26    air_layer = ba.Layer(m_ambience)
27    intermediate_layer = ba.Layer(m_interm_layer, 30.*nm)
28    intermediate_layer.addLayout(particle_layout)
29    substrate_layer = ba.Layer(m_substrate, 0)
30
31    multi_layer = ba.MultiLayer()
32    multi_layer.addLayer(air_layer)
33    multi_layer.addLayer(intermediate_layer)
34    multi_layer.addLayer(substrate_layer)
35    return multi_layer
36
37
38def get_simulation():
39    """
40    Returns a GISAXS simulation.
41    """
42    simulation = ba.GISASSimulation()
43    simulation.setSample(get_sample())
44    simulation.setDetectorParameters(200, -1*deg, +1*deg, 200, 0*deg, +2*deg)
45    simulation.setBeamParameters(1.5*angstrom, 0.15*deg, 0.0*deg)
46    return simulation
47
48
49def run_simulation():
50    """
51    Runs simulation and returns intensity map.
52    """
53    simulation = get_simulation()
54    simulation.setSample(get_sample())
55    simulation.runSimulation()
56    return simulation.getIntensityData()
57
58
59if __name__ == '__main__':
60    result = run_simulation()
61    ba.plot_intensity_data(result)
