Correlated roughness

Scattering from a multilayered sample with correlated roughness.

  • The sample is composed of a substrate on which is sitting a stack of layers. These layers consist in a repetition of 5 times two different superimposed layers (from bottom to top):
    • layer A: 2.5 nm thick with a real refractive index n = 1-5e-6.
    • layer B: 5 nm thick with a real refractive index n = 1-1e-5.
  • There is no added particle.
  • All layers present the same type of roughness on the top surface, which is characterized by:
    • a rms roughness of the interfaces σ =1 nm,
    • a Hurst parameter H equal to 0.3,
    • a lateral correlation length ξ of 5 nm,
    • a cross correlation length ξ⊥ equal to 1e-4 nm.
  • The incident beam is characterized by a wavelength of 1 Å.
  • The incident angles are αi = 0.2° and Φi = 0°.

 2MultiLayer with correlated roughness
 3"""
 4import bornagain as ba
 5from bornagain import deg, angstrom, nm
 6
 7
 8def get_sample():
 9    """
10    Returns a sample with two layers on a substrate, with correlated roughnesses.
11    """
12    # defining materials
13    m_ambience = ba.HomogeneousMaterial("ambience", 0.0, 0.0)
14    m_part_a = ba.HomogeneousMaterial("PartA", 5e-6, 0.0)
15    m_part_b = ba.HomogeneousMaterial("PartB", 10e-6, 0.0)
16    m_substrate = ba.HomogeneousMaterial("substrate", 15e-6, 0.0)
17
18    # defining layers
19    l_ambience = ba.Layer(m_ambience)
20    l_part_a = ba.Layer(m_part_a, 2.5*nm)
21    l_part_b = ba.Layer(m_part_b, 5.0*nm)
22    l_substrate = ba.Layer(m_substrate)
23
24    roughness = ba.LayerRoughness()
25    roughness.setSigma(1.0*nm)
26    roughness.setHurstParameter(0.3)
27    roughness.setLatteralCorrLength(5.0*nm)
28
29    my_sample = ba.MultiLayer()
30
31    # adding layers
32    my_sample.addLayer(l_ambience)
33
34    n_repetitions = 5
35    for i in range(n_repetitions):
36        my_sample.addLayerWithTopRoughness(l_part_a, roughness)
37        my_sample.addLayerWithTopRoughness(l_part_b, roughness)
38
39    my_sample.addLayerWithTopRoughness(l_substrate, roughness)
40    my_sample.setCrossCorrLength(10*nm)
41
42    print(my_sample.treeToString())
43
44    return my_sample
45
46
47def get_simulation():
48    """
49    Characterizing the input beam and output detector
50    """
51    simulation = ba.GISASSimulation()
52    simulation.setDetectorParameters(200, -0.5*deg, 0.5*deg,
53                                     200, 0.0*deg, 1.0*deg)
54    simulation.setBeamParameters(1.0*angstrom, 0.2*deg, 0.0*deg)
55    simulation.setBeamIntensity(5e11)
56    return simulation
57
58
59def run_simulation():
60    """
61    Runs simulation and returns intensity map.
62    """
63    simulation = get_simulation()
64    simulation.setSample(get_sample())
65    simulation.runSimulation()
66    return simulation.getIntensityData()
67
68
69if __name__ == '__main__':
70    result = run_simulation()
71    ba.plot_intensity_data(result)
